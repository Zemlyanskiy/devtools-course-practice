#############################################
##### Testing
#############################################

enable_testing()

set(prefix "${MODULE}")

add_test(
    NAME ${prefix}_can_Run
    COMMAND ${APPLICATION}
)

add_test(
    NAME ${prefix}_can_Print_Help_Without_Arguments
    COMMAND ${APPLICATION}
)
set_tests_properties (${prefix}_can_Print_Help_Without_Arguments
  PROPERTIES PASS_REGULAR_EXPRESSION "This is a matrix calculator application")

add_test(
    NAME ${prefix}_can_Check_Number_Of_Arguments
    COMMAND ${APPLICATION} 5.4 5.2 + 4.7
)
set_tests_properties (${prefix}_can_Check_Number_Of_Arguments
  PROPERTIES PASS_REGULAR_EXPRESSION "ERROR: Should be not more than 3 arguments.")
add_test(
    NAME ${prefix}_can_Detect_Wrong_Number_Format_Double
    COMMAND ${APPLICATION} 10f 3 +
)
set_tests_properties (${prefix}_can_Detect_Wrong_Number_Format_Double
  PROPERTIES PASS_REGULAR_EXPRESSION "Wrong number format!")
add_test(
    NAME ${prefix}_can_Detect_Wrong_Operation_Format
    COMMAND ${APPLICATION} 10.3 3.5 d
)
set_tests_properties (${prefix}_can_Wrong_Operation_Format
  PROPERTIES PASS_REGULAR_EXPRESSION "Wrong operation format!")
add_test(
    NAME ${prefix}_can_Search_Det_Of_Sum
    COMMAND ${APPLICATION} 10.3 3.5 +
)
set_tests_properties (${prefix}_can_Search_Det_Of_Sum
  PROPERTIES PASS_REGULAR_EXPRESSION "Det of sum = ")
add_test(
    NAME ${prefix}_can_Search_Det_Of_Dif
    COMMAND ${APPLICATION} 10.3 3.5 -
)
set_tests_properties (${prefix}_can_Search_Det_Of_Dif
  PROPERTIES PASS_REGULAR_EXPRESSION "Det of difference = ")
add_test(
    NAME ${prefix}_can_Search_Det_Of_Mul
    COMMAND ${APPLICATION} 10.3 3.5 *
)
set_tests_properties (${prefix}_can_Search_Det_Of_Mul
  PROPERTIES PASS_REGULAR_EXPRESSION "Det of multiplication = ")
